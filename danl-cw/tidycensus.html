Basic usage of tidycensus

age20 <- get_decennial(geography = "state", 
                       variables = "P13_001N", 
                       year = 2020,
                       sumfile = "dhc")

datatable(age20)
age20 %>%
  ggplot(aes(x = value, y = reorder(NAME, value))) + 
  geom_point()
